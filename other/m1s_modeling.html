def create_seller(seller_tier, period, seller_count, index):
    # print('Creating seller from type {} for period of {}'.format(seller_tier, period))
    
    global faker
    global seller_tiers_df
    # global generated_seller_df
    global seller_ls
    
    price = seller_tiers_df.loc[seller_tier]['price'][random.randint(0, len(seller_tiers_df.loc[seller_tier]['price']) - 1)]
    
    # print('Index: {}'.format(index))
    # day = int(round(((index / 100) * seller_count) * 28 + 1, 0))
    day = int(round((index / seller_count * 27)+1))
    
    # print('Day: {}'.format(day))
    
    created_at = '{}-{:02d}'.format(period, day)
    
    seller_dt = {
        '_id' : [
            uuid.uuid4()
        ],
        'createdAt' : [
            created_at
        ],
        'period' : [
            period
        ],
        'name' : [
            faker.name()
        ],
        'tier' : [
            seller_tier
        ],
        'price' : [
            price
        ],
        'profession' : [
            faker.job()
        ]
    }    
    
    seller_df = pd.DataFrame(data=seller_dt)
    seller_ls.append(seller_df)
    # generated_seller_df = generated_seller_df.append(seller_df, ignore_index=True)

def create_order(period, order_count, index):
    # print('Creating order {} for {} period of total of {} orders'.format(index, period, order_count))
    
    global faker
    global order_ls
    
    seller = get_seller_by_period(period)
    day = int(round((index / order_count * 27)+1))
    created_at = '{}-{:02d}'.format(period, day)  
    
    price = seller['price'].values[0]
    
    gross_profit = price * split_df.loc['Marketplace']['rate']
    amount_for_psp = price * marketplace_split_df.loc['PSP']['rate']
    amount_for_taxes = gross_profit * taxes_df['rate'].sum()
    amount_for_marketplace = gross_profit - amount_for_psp - amount_for_taxes

    gross_talent = price * split_df.loc['Seller']['rate']
    amount_for_talent = gross_talent * seller_split_df.loc['Seller']['rate']
    amount_for_donation = gross_talent * seller_split_df.loc['Donation']['rate']
    
    order_dt = {
        '_id' : [
            uuid.uuid4()
        ],
        'createdAt' : [
            created_at
        ],
        'period' : [
            period
        ],
        'consumerName' : [
            faker.name()
        ],
        'seller' : [
            seller['_id'].values[0]
        ],
        'price' : [
            price
        ],
        'grossProfit' : [
            gross_profit
        ],
        'amountForPSP' : [
            amount_for_psp
        ],
        'amountForTaxes' : [
            amount_for_taxes
        ],
        'amountForMarketplace' : [
            amount_for_marketplace
        ],
        'grossTalent' : [
            gross_talent
        ],
        'amountForTalent' : [
            amount_for_talent
        ],
        'amountForDonation' : [
            amount_for_donation
        ]
    }    
    
    order_df = pd.DataFrame(data=order_dt)
    # generated_order_df = generated_order_df.append(order_df, ignore_index=True)
    order_ls.append(order_df)

def get_seller_by_period(period):
    
    global generated_seller_df
    
    days_in_period = monthrange(int(period.split('-')[0]), int(period.split('-')[1]))[1]    
    valid_sellers_df = generated_seller_df[generated_seller_df['createdAt'] < pd.to_datetime('{}-{}'.format(period, days_in_period))]
    
    return valid_sellers_df.sample(n=1)

def create_orders(seller_count, period):
    order_count = seller_count * consumer_to_seller_rate
    # print('Creating {} orders for {} users under period of {}'.format(order_count, seller_count, period))
    
    i = 0
    while i < order_count:
        create_order(period, order_count, i)
        i += 1
